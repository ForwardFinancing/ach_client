def self.soap_client
  self._soap_client ||= Savon.client(wsdl: self.wsdl) do
    # Lets us use symbols as keys without Savon changing the case
    # { 'Key' => 'Value' } == { Key: 'Value' }
    convert_request_keys_to :none
  end
end
